227 void ApplicationState::multiplySleepInterval(double m_) {
228  boost::mutex::scoped_lock lock(internal_data_synchronization_mutex);
229  // Special case: leave 1, keeping the integer rounding of (>) ? : operator.
230  if (application_sleep_interval < 2)
231  application_sleep_interval = 2;
232  else
233  application_sleep_interval = application_sleep_interval * m_;
234  // Truncate sleep interval - with rounding to integers.
235  application_sleep_interval = ((application_sleep_interval > 10000000) ? 10000000 : application_sleep_interval);
236  LOG(LSTATUS) << "Setting sleep interval to " << application_sleep_interval << " [us]";
237 }
